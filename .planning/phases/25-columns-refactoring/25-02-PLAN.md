    ```tsx

    import type { TaskListColumn } from './types';

    export const BUILT_IN_COLUMN_WIDTHS: Record<string, number> = {
      number: 40,
      name: 200,
      startDate: 90,
      endDate: 90,
      duration: 60,
      progress: 50,
      dependencies: 120,
      actions: 80,
    };

    export function createBuiltInColumns<TTask extends Task>(): TaskListColumn<TTask>[] {
      return [
        { id: 'number', header: '№', width: BUILT_IN_COLUMN_WIDTHS.number, renderCell: () => null },
        { id: 'name', header: 'Имя', width: BUILT_IN_COLUMN_WIDTHS.name, renderCell: () => null },
        { id: 'startDate', header: 'Начало', width: BUILT_IN_COLUMN_WIDTHS.startDate, renderCell: () => null },
        { id: 'endDate', header: 'Окончание', width: BUILT_IN_COLUMN_WIDTHS.endDate, renderCell: () => null },
        { id: 'duration', header: 'Дн.', width: BUILT_IN_COLUMN_WIDTHS.duration, renderCell: () => null },
        { id: 'progress', header: '%', width: BUILT_IN_COLUMN_WIDTHS.progress, renderCell: () => null },
        { id: 'dependencies', header: null, width: BUILT_IN_COLUMN_WIDTHS.dependencies, renderCell: () => null },
        { id: 'actions', header: '', width: BUILT_IN_COLUMN_WIDTHS.actions, renderCell: () => null },
      ];
    }
    ```
