    ```typescript
    const handleTaskChange = useCallback((taskOrUpdater: Task | ((tasks: Task[]) => Task[])) => {
      onChange?.((currentTasks) => {
        const updatedTasks = typeof taskOrUpdater === 'function'
          ? taskOrUpdater(currentTasks)
          : currentTasks.map(t => t.id === taskOrUpdater.id ? taskOrUpdater : t);

        // For each updated task, check if it has a parent
        const parentIdsToUpdate = new Set<string>();
        updatedTasks.forEach(task => {
          if (task.parentId) parentIdsToUpdate.add(task.parentId);
        });

        // Update parent dates for each affected parent
        let finalTasks = updatedTasks;
        parentIdsToUpdate.forEach(parentId => {
          const newDates = computeParentDates(parentId, finalTasks);
          finalTasks = finalTasks.map(t =>
            t.id === parentId
              ? { ...t, startDate: newDates.startDate, endDate: newDates.endDate }
              : t
          );
        });

        return finalTasks;
      });
    }, [onChange]);
    ```
